228LSM6DSV16XStatusTypeDef LSM6DSV16X::begin()
229{
230 if (spi) {
231 // Configure CS pin
232 //pinMode(cs_pin, OUTPUT);
233 //digitalWrite(cs_pin, HIGH);
234 }
235
236 /* Enable register address auto-increment during multiple byte access with a serial interface. */
237 if (auto_increment_set(PROPERTY_ENABLE) != LSM6DSV16X_OK) {
238 return LSM6DSV16X_ERROR;
239 }
240
241 /* Enable Block Data Update (BDU). */
242 if (Enable_Block_Data_Update() != LSM6DSV16X_OK) {
243 return LSM6DSV16X_ERROR;
244 }
245
246 /* FIFO mode selection. */
247 if (fifo_mode_set(LSM6DSV16X_BYPASS_MODE) != LSM6DSV16X_OK) {
248 return LSM6DSV16X_ERROR;
249 }
250
251 /* Select default output data rate for accelerometer. */
252 acc_odr = LSM6DSV16X_ODR_AT_120Hz;
253
254 /* Output data rate selection for accelerometer - power down. */
255 if (xl_data_rate_set(LSM6DSV16X_ODR_OFF) != LSM6DSV16X_OK) {
256 return LSM6DSV16X_ERROR;
257 }
258
259 /* Full scale selection for accelerometer. */
260 if (xl_full_scale_set(LSM6DSV16X_2g) != LSM6DSV16X_OK) {
261 return LSM6DSV16X_ERROR;
262 }
263
264 /* Select default output data rate for gyroscope. */
265 gyro_odr = LSM6DSV16X_ODR_AT_120Hz;
266
267 /* Output data rate selection for gyroscope - power down. */
268 if (gy_data_rate_set(LSM6DSV16X_ODR_OFF) != LSM6DSV16X_OK) {
269 return LSM6DSV16X_ERROR;
270 }
271
272 /* Full scale selection for gyroscope. */
273 if (gy_full_scale_set(LSM6DSV16X_2000dps) != LSM6DSV16X_OK) {
274 return LSM6DSV16X_ERROR;
275 }
276
277 initialized = 1U;
278
279 return LSM6DSV16X_OK;
280}

461 break;
462 case 6:
463 newMode = LSM6DSV16X_STREAM_MODE;
464 break;
465 case 7:
466 newMode = LSM6DSV16X_BYPASS_TO_FIFO_MODE;
467 break;
468 default:
469 return LSM6DSV16X_ERROR;
470 }
471 fifo_mode = newMode;
472 if (fifo_mode_set(fifo_mode) != LSM6DSV16X_OK) {
473 return LSM6DSV16X_ERROR;
474 }
475
476 return LSM6DSV16X_OK;

760LSM6DSV16XStatusTypeDef LSM6DSV16X::Enable_Wake_Up_Detection(LSM6DSV16X_SensorIntPin_t IntPin)
761{
762 LSM6DSV16XStatusTypeDef ret = LSM6DSV16X_OK;
763 lsm6dsv16x_md1_cfg_t val1;
764 lsm6dsv16x_md2_cfg_t val2;
765 lsm6dsv16x_functions_enable_t functions_enable;
766
767 /* Output Data Rate selection */
768 if (Set_X_ODR(480) != LSM6DSV16X_OK) {
769 return LSM6DSV16X_ERROR;
770 }
771
772 /* Full scale selection */
773 if (Set_X_FS(2) != LSM6DSV16X_OK) {
774 return LSM6DSV16X_ERROR;
775 }
776
777 /* Set wake-up threshold */
778 if (Set_Wake_Up_Threshold(63) != LSM6DSV16X_OK) {
779 return LSM6DSV16X_ERROR;
780 }
781
782 /* Set wake-up duration */
783 if (Set_Wake_Up_Duration(0) != LSM6DSV16X_OK) {
784 return LSM6DSV16X_ERROR;
785 }
786
787 /* Enable wake-up event on either INT1 or INT2 pin */
788 switch (IntPin) {
789 case LSM6DSV16X_INT1_PIN:
790 if (readRegister(LSM6DSV16X_MD1_CFG, (uint8_t *)&val1, 1) != LSM6DSV16X_OK) {
791 return LSM6DSV16X_ERROR;
792 }
793
794 val1.int1_wu = PROPERTY_ENABLE;
795
796 if (writeRegister(LSM6DSV16X_MD1_CFG, (uint8_t *)&val1, 1) != LSM6DSV16X_OK) {
797 return LSM6DSV16X_ERROR;
798 }
799
800 if (readRegister(LSM6DSV16X_FUNCTIONS_ENABLE, (uint8_t *)&functions_enable, 1) != LSM6DSV16X_OK) {
801 return LSM6DSV16X_ERROR;
802 }
803
804 functions_enable.interrupts_enable = PROPERTY_ENABLE;
805
806 if (writeRegister(LSM6DSV16X_FUNCTIONS_ENABLE, (uint8_t *)&functions_enable, 1) != LSM6DSV16X_OK) {
807 return LSM6DSV16X_ERROR;
808 }
809 break;
810
811 case LSM6DSV16X_INT2_PIN:
812 if (readRegister(LSM6DSV16X_MD2_CFG, (uint8_t *)&val2, 1) != LSM6DSV16X_OK) {
813 return LSM6DSV16X_ERROR;
814 }
815
816 val2.int2_wu = PROPERTY_ENABLE;
817
818 if (writeRegister(LSM6DSV16X_MD2_CFG, (uint8_t *)&val2, 1) != LSM6DSV16X_OK) {
819 return LSM6DSV16X_ERROR;
820 }
821
822 if (readRegister(LSM6DSV16X_FUNCTIONS_ENABLE, (uint8_t *)&functions_enable, 1) != LSM6DSV16X_OK) {
823 return LSM6DSV16X_ERROR;
824 }
825
826 functions_enable.interrupts_enable = PROPERTY_ENABLE;
827
828 if (writeRegister(LSM6DSV16X_FUNCTIONS_ENABLE, (uint8_t *)&functions_enable, 1) != LSM6DSV16X_OK) {
829 return LSM6DSV16X_ERROR;
830 }
831 break;
832
833 default:
834 ret = LSM6DSV16X_ERROR;
835 break;
836 }
837
838 return ret;

4108LSM6DSV16XStatusTypeDef LSM6DSV16X::npy_halfbits_to_floatbits(uint16_t h, uint32_t *f)
4109{
4110 uint16_t h_exp, h_sig;
4111 uint32_t f_sgn, f_exp, f_sig;
4112
4113 h_exp = (h & 0x7c00u);
4114 f_sgn = ((uint32_t)h & 0x8000u) << 16;
4115 switch (h_exp) {
4116 case 0x0000u: /* 0 or subnormal */
4117 h_sig = (h & 0x03ffu);
4118 /* Signed zero */
4119 if (h_sig == 0) {
4120 *f = f_sgn;
4121 return LSM6DSV16X_OK;
4122 }
4123 /* Subnormal */
4124 h_sig <<= 1;
4125 while ((h_sig & 0x0400u) == 0) {
4126 h_sig <<= 1;
4127 h_exp++;
4128 }
4129 f_exp = ((uint32_t)(127 - 15 - h_exp)) << 23;
4130 f_sig = ((uint32_t)(h_sig & 0x03ffu)) << 13;
4131 *f = f_sgn + f_exp + f_sig;
4132 return LSM6DSV16X_OK;
4133 case 0x7c00u: /* inf or NaN */
4134 /* All-ones exponent and a copy of the significand */
4135 *f = f_sgn + 0x7f800000u + (((uint32_t)(h & 0x03ffu)) << 13);
4136 return LSM6DSV16X_OK;
4137 default: /* normalized */
4138 /* Just need to adjust the exponent and shift */
4139 *f = f_sgn + (((uint32_t)(h & 0x7fffu) + 0x1c000u) << 13);
4140 return LSM6DSV16X_OK;
4141 }
4142}

5266int32_t LSM6DSV16X::sflp_game_gbias_set(lsm6dsv16x_sflp_gbias_t *val)
5267{

5269 lsm6dsv16x_emb_func_exec_status_t emb_func_sts;
5270 lsm6dsv16x_data_ready_t drdy;
5271 lsm6dsv16x_xl_full_scale_t xl_fs;
5272 lsm6dsv16x_ctrl10_t ctrl10;
5273 uint8_t master_config;
5274 uint8_t emb_func_en_saved[2];
5275 uint8_t conf_saved[2];
5276 uint8_t reg_zero[2] = {0x0, 0x0};
5277 uint16_t gbias_hf[3];
5278 float_t k = 0.005f;
5279 int16_t xl_data[3];
5280 int32_t data_tmp;
5281 uint8_t *data_ptr = (uint8_t *)&data_tmp;
5282 uint8_t i, j;
5283 int32_t ret;
5284
5285 ret = sflp_data_rate_get(&sflp_odr);
5286 if (ret != 0) {
5287 return ret;
5288 }
5289
5290 /* Calculate k factor */
5291 switch (sflp_odr) {
5292 default:
5293 case LSM6DSV16X_SFLP_15Hz:
5294 k = 0.04f;
5295 break;
5296 case LSM6DSV16X_SFLP_30Hz:
5297 k = 0.02f;
5298 break;
5299 case LSM6DSV16X_SFLP_60Hz:
5300 k = 0.01f;
5301 break;
5302 case LSM6DSV16X_SFLP_120Hz:
5303 k = 0.005f;
5304 break;
5305 case LSM6DSV16X_SFLP_240Hz:
5306 k = 0.0025f;
5307 break;
5308 case LSM6DSV16X_SFLP_480Hz:
5309 k = 0.00125f;
5310 break;
5311 }
5312
5313 /* compute gbias as half precision float in order to be put in embedded advanced feature register */
5314 gbias_hf[0] = npy_float_to_half(val->gbias_x * (3.14159265358979323846f / 180.0f) / k);
5315 gbias_hf[1] = npy_float_to_half(val->gbias_y * (3.14159265358979323846f / 180.0f) / k);
5316 gbias_hf[2] = npy_float_to_half(val->gbias_z * (3.14159265358979323846f / 180.0f) / k);
5317
5318 /* Save sensor configuration and set high-performance mode (if the sensor is in power-down mode, turn it on) */
5319 ret += readRegister(LSM6DSV16X_CTRL1, conf_saved, 2);
5320 ret += xl_mode_set(LSM6DSV16X_XL_HIGH_PERFORMANCE_MD);
5321 ret += gy_mode_set(LSM6DSV16X_GY_HIGH_PERFORMANCE_MD);
5322 if (((uint8_t)conf_saved[0] & 0x0FU) == (uint8_t)LSM6DSV16X_ODR_OFF) {
5323 ret += xl_data_rate_set(LSM6DSV16X_ODR_AT_120Hz);
5324 }
5325
5326 /* Make sure to turn the sensor-hub master off */
5327 ret += sh_master_get(&master_config);
5328 ret += sh_master_set(0);
5329
5330 /* disable algos */
5331 ret += mem_bank_set(LSM6DSV16X_EMBED_FUNC_MEM_BANK);
5332 ret += readRegister(LSM6DSV16X_EMB_FUNC_EN_A, emb_func_en_saved, 2);
5333 ret += writeRegister(LSM6DSV16X_EMB_FUNC_EN_A, reg_zero, 2);
5334 do {
5335 ret += readRegister(LSM6DSV16X_EMB_FUNC_EXEC_STATUS,
5336 (uint8_t *)&emb_func_sts, 1);
5337 } while (emb_func_sts.emb_func_endop != 1U);
5338 ret += mem_bank_set(LSM6DSV16X_MAIN_MEM_BANK);
5339
5340 // enable gbias setting
5341 ret += readRegister(LSM6DSV16X_CTRL10, (uint8_t *)&ctrl10, 1);
5342 ctrl10.emb_func_debug = 1;
5343 ret += writeRegister(LSM6DSV16X_CTRL10, (uint8_t *)&ctrl10, 1);
5344
5345 /* enable algos */
5346 ret += mem_bank_set(LSM6DSV16X_EMBED_FUNC_MEM_BANK);
5347 emb_func_en_saved[0] |= 0x02U; /* force SFLP GAME en */
5348 ret += writeRegister(LSM6DSV16X_EMB_FUNC_EN_A, emb_func_en_saved,
5349 2);
5350 ret += mem_bank_set(LSM6DSV16X_MAIN_MEM_BANK);
5351
5352 ret += xl_full_scale_get(&xl_fs);
5353
5354 /* Read XL data */
5355 do {
5356 ret += flag_data_ready_get(&drdy);
5357 } while (drdy.drdy_xl != 1U);
5358 ret += acceleration_raw_get(xl_data);
5359
5360 /* force sflp initialization */
5361 ret += mem_bank_set(LSM6DSV16X_SENSOR_HUB_MEM_BANK);
5362 for (i = 0; i < 3U; i++) {
5363 j = 0;
5364 data_tmp = (int32_t)xl_data[i];
5365 data_tmp <<= xl_fs; // shift based on current fs
5366 ret += writeRegister(LSM6DSV16X_SENSOR_HUB_1 + 3U * i,
5367 &data_ptr[j++], 1);
5368 ret += writeRegister(LSM6DSV16X_SENSOR_HUB_2 + 3U * i,
5369 &data_ptr[j++], 1);
5370 ret += writeRegister(LSM6DSV16X_SENSOR_HUB_3 + 3U * i, &data_ptr[j],
5371 1);
5372 }
5373 for (i = 0; i < 3U; i++) {
5374 j = 0;
5375 data_tmp = 0;
5376 ret += writeRegister(LSM6DSV16X_SENSOR_HUB_10 + 3U * i,
5377 &data_ptr[j++], 1);
5378 ret += writeRegister(LSM6DSV16X_SENSOR_HUB_11 + 3U * i,
5379 &data_ptr[j++], 1);
5380 ret += writeRegister(LSM6DSV16X_SENSOR_HUB_12 + 3U * i, &data_ptr[j],
5381 1);
5382 }
5383 ret += mem_bank_set(LSM6DSV16X_MAIN_MEM_BANK);
5384
5385 // wait end_op (and at least 30 us)
5386 wait_us(40);
5387
5388 ret += mem_bank_set(LSM6DSV16X_EMBED_FUNC_MEM_BANK);
5389 do {
5390 ret += readRegister(LSM6DSV16X_EMB_FUNC_EXEC_STATUS,
5391 (uint8_t *)&emb_func_sts, 1);
5392 } while (emb_func_sts.emb_func_endop != 1U);
5393 ret += mem_bank_set(LSM6DSV16X_MAIN_MEM_BANK);
5394
5395 /* write gbias in embedded advanced features registers */
5396 ret += ln_pg_write(LSM6DSV16X_SFLP_GAME_GBIASX_L,
5397 (uint8_t *)gbias_hf, 6);
5398
5399 /* reload previous sensor configuration */
5400 ret += writeRegister(LSM6DSV16X_CTRL1, conf_saved, 2);
5401
5402 // disable gbias setting
5403 ctrl10.emb_func_debug = 0;
5404 ret += writeRegister(LSM6DSV16X_CTRL10, (uint8_t *)&ctrl10, 1);
5405
5406 /* reload previous master configuration */
5407 ret += sh_master_set(master_config);
5408
5409 return ret;
5410}

5461 float_t f;
5462 uint32_t fbits;

5467
5468uint16_t LSM6DSV16X::npy_floatbits_to_halfbits(uint32_t f)
5469{
5470 uint32_t f_exp, f_sig;
5471 uint16_t h_sgn, h_exp, h_sig;
5472
5473 h_sgn = (uint16_t)((f & 0x80000000u) >> 16);
5474 f_exp = (f & 0x7f800000u);
5475
5476 /* Exponent overflow/NaN converts to signed inf/NaN */
5477 if (f_exp >= 0x47800000u) {
5478 if (f_exp == 0x7f800000u) {
5479 /* Inf or NaN */
5480 f_sig = (f & 0x007fffffu);
5481 if (f_sig != 0U) {
5482 /* NaN - propagate the flag in the significand... */
5483 uint16_t ret = (uint16_t)(0x7c00u + (f_sig >> 13));
5484 /* ...but make sure it stays a NaN */
5485 if (ret == 0x7c00u) {
5486 ret++;
5487 }
5488 return h_sgn + ret;
5489 } else {
5490 /* signed inf */
5491 return (uint16_t)(h_sgn + 0x7c00u);
5492 }
5493 } else {
5494 /* overflow to signed inf */
5495#if NPY_HALF_GENERATE_OVERFLOW
5496 npy_set_floatstatus_overflow();
5497#endif
5498 return (uint16_t)(h_sgn + 0x7c00u);
5499 }
5500 }
5501
5502 /* Exponent underflow converts to a subnormal half or signed zero */
5503 if (f_exp <= 0x38000000u) {
5504 /*
5505 * Signed zeros, subnormal floats, and floats with small
5506 * exponents all convert to signed zero half-floats.
5507 */
5508 if (f_exp < 0x33000000u) {
5509#if NPY_HALF_GENERATE_UNDERFLOW
5510 /* If f != 0, it underflowed to 0 */
5511 if ((f & 0x7fffffff) != 0) {
5512 npy_set_floatstatus_underflow();
5513 }
5514#endif
5515 return h_sgn;
5516 }
5517 /* Make the subnormal significand */
5518 f_exp >>= 23;
5519 f_sig = (0x00800000u + (f & 0x007fffffu));
5520#if NPY_HALF_GENERATE_UNDERFLOW
5521 /* If it's not exactly represented, it underflowed */
5522 if ((f_sig & (((uint32_t)1 << (126 - f_exp)) - 1)) != 0) {
5523 npy_set_floatstatus_underflow();
5524 }
5525#endif
5526 /*
5527 * Usually the significand is shifted by 13. For subnormals an
5528 * additional shift needs to occur. This shift is one for the largest
5529 * exponent giving a subnormal `f_exp = 0x38000000 >> 23 = 112`, which
5530 * offsets the new first bit. At most the shift can be 1+10 bits.
5531 */
5532 f_sig >>= (113U - f_exp);
5533 /* Handle rounding by adding 1 to the bit beyond half precision */
5534#if NPY_HALF_ROUND_TIES_TO_EVEN
5535 /*
5536 * If the last bit in the half significand is 0 (already even), and
5537 * the remaining bit pattern is 1000...0, then we do not add one
5538 * to the bit after the half significand. However, the (113 - f_exp)
5539 * shift can lose up to 11 bits, so the || checks them in the original.
5540 * In all other cases, we can just add one.
5541 */
5542 if (((f_sig & 0x00003fffu) != 0x00001000u) || (f & 0x000007ffu)) {
5543 f_sig += 0x00001000u;
5544 }
5545#else
5546 f_sig += 0x00001000u;
5547#endif
5548 h_sig = (uint16_t)(f_sig >> 13);
5549 /*
5550 * If the rounding causes a bit to spill into h_exp, it will
5551 * increment h_exp from zero to one and h_sig will be zero.
5552 * This is the correct result.
5553 */
5554 return (uint16_t)(h_sgn + h_sig);
5555 }
5556
5557 /* Regular case with no overflow or underflow */
5558 h_exp = (uint16_t)((f_exp - 0x38000000u) >> 13);
5559 /* Handle rounding by adding 1 to the bit beyond half precision */
5560 f_sig = (f & 0x007fffffu);
5561#if NPY_HALF_ROUND_TIES_TO_EVEN
5562 /*
5563 * If the last bit in the half significand is 0 (already even), and
5564 * the remaining bit pattern is 1000...0, then we do not add one
5565 * to the bit after the half significand. In all other cases, we do.
5566 */
5567 if ((f_sig & 0x00003fffu) != 0x00001000u) {
5568 f_sig += 0x00001000u;
5569 }
5570#else
5571 f_sig += 0x00001000u;
5572#endif
5573 h_sig = (uint16_t)(f_sig >> 13);
5574 /*
5575 * If the rounding causes a bit to spill into h_exp, it will
5576 * increment h_exp by one and h_sig will be zero. This is the
5577 * correct result. h_exp may increment to 15, at greatest, in
5578 * which case the result overflows to a signed inf.
5579 */
5580#if NPY_HALF_GENERATE_OVERFLOW
5581 h_sig += h_exp;
5582 if (h_sig == 0x7c00u) {
5583 npy_set_floatstatus_overflow();
5584 }
5585 return h_sgn + h_sig;
5586#else
5587 return h_sgn + h_exp + h_sig;
5588#endif
5589}

5766int32_t LSM6DSV16X::ln_pg_write(uint16_t address, uint8_t *buf, uint8_t len)
5767{
5768 lsm6dsv16x_page_address_t page_address;
5769 lsm6dsv16x_page_sel_t page_sel;
5770 lsm6dsv16x_page_rw_t page_rw;
5771 uint8_t msb;
5772 uint8_t lsb;
5773 int32_t ret;
5774 uint8_t i ;
5775
5776 msb = ((uint8_t)(address >> 8) & 0x0FU);
5777 lsb = (uint8_t)address & 0xFFU;
5778
5779 ret = mem_bank_set(LSM6DSV16X_EMBED_FUNC_MEM_BANK);
5780 if (ret != 0) {
5781 return ret;
5782 }
5783
5784 /* set page write */
5785 ret += readRegister(LSM6DSV16X_PAGE_RW, (uint8_t *)&page_rw, 1);
5786 page_rw.page_read = PROPERTY_DISABLE;
5787 page_rw.page_write = PROPERTY_ENABLE;
5788 ret += writeRegister(LSM6DSV16X_PAGE_RW, (uint8_t *)&page_rw, 1);
5789 if (ret != 0) {
5790 goto exit;
5791 }
5792
5793 /* select page */
5794 ret += readRegister(LSM6DSV16X_PAGE_SEL, (uint8_t *)&page_sel, 1);
5795 page_sel.page_sel = msb;
5796 page_sel.not_used0 = 1; // Default value
5797 ret += writeRegister(LSM6DSV16X_PAGE_SEL, (uint8_t *)&page_sel, 1);
5798 if (ret != 0) {
5799 goto exit;
5800 }
5801
5802 /* set page addr */
5803 page_address.page_addr = lsb;
5804 ret += writeRegister(LSM6DSV16X_PAGE_ADDRESS,
5805 (uint8_t *)&page_address, 1);
5806 if (ret != 0) {
5807 goto exit;
5808 }
5809
5810 for (i = 0; ((i < len) && (ret == 0)); i++) {
5811 ret += writeRegister(LSM6DSV16X_PAGE_VALUE, &buf[i], 1);
5812 if (ret != 0) {
5813 goto exit;
5814 }
5815
5816 lsb++;
5817
5818 /* Check if page wrap */
5819 if (((lsb & 0xFFU) == 0x00U) && (ret == 0)) {
5820 msb++;
5821 ret += readRegister(LSM6DSV16X_PAGE_SEL, (uint8_t *)&page_sel, 1);
5822 if (ret != 0) {
5823 goto exit;
5824 }
5825
5826 page_sel.page_sel = msb;
5827 page_sel.not_used0 = 1; // Default value
5828 ret += writeRegister(LSM6DSV16X_PAGE_SEL, (uint8_t *)&page_sel, 1);
5829 if (ret != 0) {
5830 goto exit;
5831 }
5832 }
5833 }
5834
5835 page_sel.page_sel = 0;
5836 page_sel.not_used0 = 1;// Default value
5837 ret += writeRegister(LSM6DSV16X_PAGE_SEL, (uint8_t *)&page_sel, 1);
5838 if (ret != 0) {
5839 goto exit;
5840 }
5841
5842 /* unset page write */
5843 ret += readRegister(LSM6DSV16X_PAGE_RW, (uint8_t *)&page_rw, 1);
5844 page_rw.page_read = PROPERTY_DISABLE;
5845 page_rw.page_write = PROPERTY_DISABLE;
5846 ret += writeRegister(LSM6DSV16X_PAGE_RW, (uint8_t *)&page_rw, 1);

5848exit:
5849 ret += mem_bank_set(LSM6DSV16X_MAIN_MEM_BANK);
